362 	/**
363 	 * @param format
364 	 * 			requested mime type 
365 	 * @param sparqlQuery
366 	 * 			string containing the SPARQL query
367 	 * @return HTTP response of evaluating the sparql query 
368 	 * @throws InternalErrorException
369 	 */
370 	private Response getSparqlResponse(final String format, String sparqlQuery) throws InternalErrorException {
371 		logger.info("SPARQL query was requested. Query: " + sparqlQuery);
372 		String user = null;
373 		Matcher userMatcher = patternUser.matcher(sparqlQuery);
374 		if (userMatcher.find()) {
375 			user = userMatcher.group("user");
376 			sparqlQuery = userMatcher.replaceAll("");
377 		}
378 		String message = null;
379 		Matcher messageMatcher = patternCommitMessage.matcher(sparqlQuery);
380 		if (messageMatcher.find()) {
381 			message = messageMatcher.group("message");
382 			sparqlQuery = messageMatcher.replaceAll("");
383 		}
384 
385 		if (patternSelectAskConstructQuery.matcher(sparqlQuery).find()) {
386 			return getSelectConstructAskResponse(sparqlQuery, format);
387 		}
388 		if (patternUpdateQuery.matcher(sparqlQuery).find()) {
389 			return getUpdateResponse(sparqlQuery, user, message, format);
390 		}
391 		if (patternCreateGraph.matcher(sparqlQuery).find()) {
392 			return getCreateGraphResponse(sparqlQuery, format);
393 		}
394 		if (patternMergeQuery.matcher(sparqlQuery).find()) {
395 			return getMergeResponse(sparqlQuery, user, message, format);
396 		}
397 		if (patternDropGraph.matcher(sparqlQuery).find()) {
398 			return getDropGraphResponse(sparqlQuery, format);
399 		}
400 		if (patternBranchOrTagQuery.matcher(sparqlQuery).find()) {
401 			return getBranchOrTagResponse(sparqlQuery, user, message, format);
402 		}
403 		throw new QueryErrorException("No R43ples query detected");
404 	}

430 	 * Provides the SPARQL Endpoint description of the original sparql endpoint
431 	 * with the additional R43ples feature (sd:feature) and replaced URIs.
432 	 * 
433 	 * @param format
434 	 *            serialisation format of the service description
435 	 * @return Extended Service Description
436 	 * @throws ClientProtocolException
437 	 * @throws IOException
438 	 */
439 	private Response getServiceDescriptionResponse(final String format) {
440 		logger.info("Service Description requested");
441 		String triples =String.format("@prefix rdf:	<http://www.w3.org/1999/02/22-rdf-syntax-ns#> . %n"
442 				+ "@prefix ns3:	<http://www.w3.org/ns/formats/> .%n"
443 				+ "@prefix sd:	<http://www.w3.org/ns/sparql-service-description#> .%n"
444 				+ "<%1$s>	rdf:type	sd:Service ;%n"
445 				+ "	sd:endpoint	<%1$s> ;%n"
446 				+ "	sd:feature	sd:r43ples ;"
447 				+ "	sd:resultFormat	ns3:SPARQL_Results_JSON ,%n"
448 				+ "		ns3:SPARQL_Results_XML ,%n"
449 				+ "		ns3:Turtle ,%n"
450 				+ "		ns3:N-Triples ,%n"
451 				+ "		ns3:N3 ,%n"
452 				+ "		ns3:RDF_XML ,%n"
453 				+ "		ns3:SPARQL_Results_CSV ,%n"
454 				+ "		ns3:RDFa ;%n"
455 				+ "	sd:supportedLanguage	sd:SPARQL10Query, sd:SPARQL11Query, sd:SPARQL11Query, sd:SPARQL11Update, sd:R43plesQuery  ;%n"
456 				+ "	sd:url	<%1$s> .%n", uriInfo.getAbsolutePath()) ;
457 		Model model = JenaModelManagement.readStringToJenaModel(triples, "TURTLE");
458 		ByteArrayOutputStream baos = new ByteArrayOutputStream();
459 		
460 		if (format.toLowerCase().contains("xml") )
461 			model.write(baos, "RDF/XML");
462 		else if (format.toLowerCase().contains("turtle") )
463 			model.write(baos, "Turtle");
464 		else if (format.toLowerCase().contains("json") )
465 			model.write(baos, "RDF/JSON");
466 		else {
467 			try {
468 				model.write(baos, format);
469 			}
470 			catch (NoWriterForLangException e) {
471 				model.write(baos, "Turtle");
472 			}
473 		}
474 		return Response.ok().entity(baos.toString()).build();
475 	}

477 	/**
478 	 * Produce the response for a SELECT, ASK or CONSTRUCT SPARQL query. can handle multiple
479 	 * graphs
480 	 * 
481 	 * @param query
482 	 *            the SPARQL query
483 	 * @param format
484 	 *            the result format
485 	 * @return the response with HTTP header for every graph (revision number
486 	 *         and MASTER revision number)
487 	 * @throws InternalErrorException 
488 	 */
489 	private Response getSelectConstructAskResponse(final String query, final String format) throws InternalErrorException {
490 		ResponseBuilder responseBuilder = Response.ok();
491 		String result;
492 		
493 		if (query.contains("OPTION r43ples:SPARQL_JOIN")) {
494 			String query_rewritten = query.replace("OPTION r43ples:SPARQL_JOIN", "");
495 			query_rewritten = SparqlRewriter.rewriteQuery(query_rewritten);
496 			result = TripleStoreInterfaceSingleton.get().executeSelectConstructAskQuery(query_rewritten, format);
497 		}
498 		else {
499 			result = getSelectConstructAskResponseClassic(query, format);
500 		}
501 		
502 		if (format.equals("text/html")){
503 			responseBuilder.entity(getHTMLResult(result, query));
504 		} else {
505 			responseBuilder.entity(result);
506 		}
507 		responseBuilder.type(format);
508 		responseBuilder.header("r43ples-revisiongraph", RevisionManagement.getResponseHeaderFromQuery(query));		
509 		return responseBuilder.build();
510 	}

529 	/**
530 	 * @param query
531 	 * @param format
532 	 * @return
533 	 * @throws InternalErrorException 
534 	 */
535 	private String getSelectConstructAskResponseClassic(final String query, final String format) throws InternalErrorException {
536 		String queryM = query;
537 
538 		Matcher m = patternSelectFromPart.matcher(queryM);
539 		while (m.find()) {
540 			String graphName = m.group("graph");
541 			String type = m.group("type");
542 			String revisionNumber = m.group("revision").toLowerCase();
543 			String newGraphName;
544 
545 			// if no revision number is declared use the MASTER as default
546 			if (revisionNumber == null) {
547 				revisionNumber = "master";
548 			}
549 			if (revisionNumber.equalsIgnoreCase("master")) {
550 				// Respond with MASTER revision - nothing to be done - MASTER revisions are already created in the named graphs
551 				newGraphName = graphName;
552 			} else {
553 				if (RevisionManagement.isBranch(graphName, revisionNumber)) {
554 					newGraphName = RevisionManagement.getReferenceGraph(graphName, revisionNumber);
555 				} else {
556 					// Respond with specified revision, therefore the revision must be generated - saved in graph <RM-TEMP-graphName>
557 					newGraphName = graphName + "-temp";
558 					RevisionManagement.generateFullGraphOfRevision(graphName, revisionNumber, newGraphName);
559 				}
560 			}
561 
562 			queryM = m.replaceFirst(type + " <" + newGraphName + ">");
563 			m = patternSelectFromPart.matcher(queryM);
564 			
565 		}
566 		String response = TripleStoreInterfaceSingleton.get().executeSelectConstructAskQuery(queryM, format);
567 		return response;
568 	}

571 	/**
572 	 * Produce the response for a INSERT or DELETE SPARQL query.
573 	 * 
574 	 * @param query
575 	 *            the SPARQL query containing the update request
576 	 * @param user
577 	 * 			  user committing the update
578 	 * @param commitMessage
579 	 * 			  message describing the commit
580 	 * @param format
581 	 *            the result format
582 	 * @return the response with HTTP header for every graph (revision number
583 	 *         and MASTER revision number)
584 	 * @throws InternalErrorException 
585 	 */
586 	private Response getUpdateResponse(final String query, final String user, final String commitMessage,
587 			final String format) throws InternalErrorException {
588 		logger.info("Update detected");
589 		
590 		// write to add and delete sets
591 		// (replace graph names in query)
592 		String queryM = query;
593 		Matcher m = patternUpdateRevision.matcher(queryM);
594 		while (m.find()) {
595 			String graphName = m.group("graph");
596 			String revisionName = m.group("revision").toLowerCase(); 	// can contain revision
597 																		// numbers or reference
598 																		// names
599 			String action = m.group("action");
600 			String data = m.group("data");
601 			if (data == null)
602 				data = "";
603 			
604 			String newRevisionNumber = RevisionManagement.getNextRevisionNumber(graphName, revisionName);
605 			String addSetGraphUri = graphName + "-delta-added-" + newRevisionNumber;
606 			String removeSetGraphUri = graphName + "-delta-removed-" + newRevisionNumber;
607 			if (!RevisionManagement.isBranch(graphName, revisionName)) {
608 				throw new InternalErrorException("Revision is not referenced by a branch");
609 			}
610 			if (action.equalsIgnoreCase("INSERT")) {
611 				queryM = m.replaceFirst(String.format("INSERT %s { GRAPH <%s>", data, addSetGraphUri));
612 			} else if (action.equalsIgnoreCase("DELETE")) {
613 				queryM = m.replaceFirst(String.format("INSERT %s { GRAPH <%s>", data, removeSetGraphUri));
614 			} else if (action.equalsIgnoreCase("WHERE")) {
615 				// TODO ersetze mit SPARQL JOIN
616 				String tempGraphName = graphName + "-temp";
617 				RevisionManagement.generateFullGraphOfRevision(graphName, revisionName, tempGraphName);
618 				queryM = m.replaceFirst(String.format("WHERE { GRAPH <%s>", tempGraphName));
619 			}
620 			m = patternUpdateRevision.matcher(queryM);
621 		}
622 		
623 		// Remove empty insert clauses which otherwise will lead to errors
624 		m= patternEmptyGraphPattern.matcher(queryM);
625 		queryM = m.replaceAll("");
626 
627 		TripleStoreInterfaceSingleton.get().executeUpdateQuery(queryM);
628 
629 		queryM = query;
630 		m = patternGraphWithRevision.matcher(queryM);
631 		while (m.find()) {
632 			String graphName = m.group("graph");
633 			String revisionName = m.group("revision").toLowerCase();	// can contain revision
634 																		// numbers or reference
635 																		// names
636 			// General variables
637 			String newRevisionNumber = RevisionManagement.getNextRevisionNumber(graphName, revisionName);
638 			String referenceFullGraph = RevisionManagement.getReferenceGraph(graphName, revisionName);
639 			String addSetGraphUri = graphName + "-delta-added-" + newRevisionNumber;
640 			String removeSetGraphUri = graphName + "-delta-removed-" + newRevisionNumber;
641 
642 			// remove doubled data
643 			// (already existing triples in add set; not existing triples in
644 			// delete set)
645 			TripleStoreInterfaceSingleton.get().executeUpdateQuery(String.format(
646 							"DELETE { GRAPH <%s> { ?s ?p ?o. } } WHERE { GRAPH <%s> { ?s ?p ?o. } }", addSetGraphUri,
647 							referenceFullGraph));
648 			TripleStoreInterfaceSingleton.get().executeUpdateQuery(String.format(
649 					"DELETE { GRAPH <%s> { ?s ?p ?o. } } WHERE { GRAPH <%s> { ?s ?p ?o. } MINUS { GRAPH <%s> { ?s ?p ?o. } } }",
650 					removeSetGraphUri, removeSetGraphUri, referenceFullGraph));
651 
652 			// merge change sets into reference graph
653 			// (copy add set to reference graph; remove delete set from reference graph)
654 			TripleStoreInterfaceSingleton.get().executeUpdateQuery(String.format(
655 						"INSERT { GRAPH <%s> { ?s ?p ?o. } } WHERE { GRAPH <%s> { ?s ?p ?o. } }",
656 						referenceFullGraph,	addSetGraphUri));
657 			TripleStoreInterfaceSingleton.get().executeUpdateQuery(String.format(
658 					"DELETE { GRAPH <%s> { ?s ?p ?o. } } WHERE { GRAPH <%s> { ?s ?p ?o. } }", 
659 					referenceFullGraph,	removeSetGraphUri));
660 
661 			// add meta information to R43ples
662 			ArrayList<String> usedRevisionNumber = new ArrayList<String>();
663 			usedRevisionNumber.add(revisionName);
664 			RevisionManagement.addMetaInformationForNewRevision(graphName, user, commitMessage, usedRevisionNumber,
665 					newRevisionNumber, addSetGraphUri, removeSetGraphUri);
666 			
667 			
668 			queryM = m.replaceAll(String.format("GRAPH <%s> ", graphName));
669 			m = patternGraphWithRevision.matcher(queryM);
670 		}
671 		
672 		ResponseBuilder responseBuilder = Response.created(URI.create(""));
673 		
674 		String result = "Query successfully executed";
675 		if (format.equals("text/html")){
676 			responseBuilder.entity(getHTMLResult(result, query));
677 		} else {
678 			responseBuilder.entity(result);
679 		}
680 		responseBuilder.type(format);
681 		responseBuilder.header("r43ples-revisiongraph", RevisionManagement.getResponseHeaderFromQuery(query));		
682 		return responseBuilder.build();
683 	}

815 	/** 
816 	 * Creates a merge between the specified branches.
817 	 * 
818 	 * Using command: MERGE GRAPH <graphURI> BRANCH "branchNameA" INTO "branchNameB"
819 	 * 
820 	 * @param sparqlQuery the SPARQL query
821 	 * @param format the result format
822 	 * @throws InternalErrorException 
823 	 */
824 	private Response getMergeResponse(final String sparqlQuery, final String user, final String commitMessage, final String format) throws InternalErrorException {
825 		ResponseBuilder responseBuilder = Response.created(URI.create(""));
826 		logger.info("Merge creation detected");
827 
828 		// Add R43ples information
829 		Matcher m = patternMergeQuery.matcher(sparqlQuery);
830 		
831 		boolean foundEntry = false;
832 		while (m.find()) {
833 			foundEntry = true;
834 			String newRevisionNumber = null;
835 			
836 			String action = m.group("action");
837 			String graphName = m.group("graph");
838 			String sdd = m.group("sdd");
839 			String sddURI = m.group("sddURI");
840 			String branchNameA = m.group("branchNameA").toLowerCase();
841 			String branchNameB = m.group("branchNameB").toLowerCase();
842 			String with = m.group("with");
843 			String triples = m.group("triples");
844 			
845 			String revisionUriA = RevisionManagement.getRevisionUri(graphName, branchNameA);
846 			String revisionUriB = RevisionManagement.getRevisionUri(graphName, branchNameB);
847 			
848 			logger.debug("action: " + action);
849 			logger.debug("graph: " + graphName);
850 			logger.debug("sdd: " + sdd);
851 			logger.debug("sddURI: " + sddURI);
852 			logger.debug("branchNameA: " + branchNameA);
853 			logger.debug("branchNameB: " + branchNameB);
854 			logger.debug("with: " + with);
855 			logger.debug("triples: " + triples);
856 			
857 			if (!RevisionManagement.checkGraphExistence(graphName)){
858 				logger.error("Graph <"+graphName+"> does not exist.");
859 				throw new InternalErrorException("Graph <"+graphName+"> does not exist.");
860 			}
861 				
862 			
863 			// Check if A and B are different revisions
864 			if (RevisionManagement.getRevisionNumber(graphName, branchNameA).equals(RevisionManagement.getRevisionNumber(graphName, branchNameB))) {
865 				// Branches are equal - throw error
866 				throw new InternalErrorException("Specified branches are equal: " + sparqlQuery);
867 			}
868 			
869 			// Check if both are terminal nodes
870 			if (!(RevisionManagement.isBranch(graphName, branchNameA) && RevisionManagement.isBranch(graphName, branchNameB))) {
871 				throw new InternalErrorException("Non terminal nodes were used: " + sparqlQuery);
872 			}
873 
874 			
875 			// Differ between MERGE query with specified SDD and without SDD			
876 			String usedSDDURI = null;
877 			if (sdd != null) {
878 				// Specified SDD
879 				usedSDDURI = sddURI;
880 			} else {
881 				// Default SDD
882 				// Query the referenced SDD
883 				String querySDD = String.format(
884 						  "PREFIX sddo: <http://eatld.et.tu-dresden.de/sddo#> %n"
885 						+ "PREFIX rmo: <http://eatld.et.tu-dresden.de/rmo#> %n"
886 						+ "SELECT ?defaultSDD %n"
887 						+ "WHERE { GRAPH <%s> {	%n"
888 						+ "	<%s> a rmo:Graph ;%n"
889 						+ "		sddo:hasDefaultSDD ?defaultSDD . %n"
890 						+ "} }", Config.revision_graph, graphName);
891 				
892 				ResultSet resultSetSDD = TripleStoreInterfaceSingleton.get().executeSelectQuery(querySDD);
893 				if (resultSetSDD.hasNext()) {
894 					QuerySolution qs = resultSetSDD.next();
895 					usedSDDURI = qs.getResource("?defaultSDD").toString();
896 				} else {
897 					throw new InternalErrorException("Error in revision graph! Selected graph <" + graphName + "> has no default SDD referenced.");
898 				}
899 			}
900 
901 			// Get the common revision with shortest path
902 			String commonRevision = MergeManagement.getCommonRevisionWithShortestPath(revisionUriA, revisionUriB);
903 			
904 			// Create the revision progress for A and B
905 			String graphNameA = graphName + "-RM-REVISION-PROGRESS-A";
906 			String graphNameB = graphName + "-RM-REVISION-PROGRESS-B";
907 			String graphNameDiff = graphName + "-RM-DIFFERENCE-MODEL";
908 			String uriA = "http://eatld.et.tu-dresden.de/branch-A";
909 			String uriB = "http://eatld.et.tu-dresden.de/branch-B";
910 			
911 			MergeManagement.createRevisionProgress(MergeManagement.getPathBetweenStartAndTargetRevision(commonRevision, revisionUriA), graphNameA, uriA);
912 			MergeManagement.createRevisionProgress(MergeManagement.getPathBetweenStartAndTargetRevision(commonRevision, revisionUriB), graphNameB, uriB);
913 			
914 			// Create difference model
915 			MergeManagement.createDifferenceTripleModel(graphName,  graphNameDiff, graphNameA, uriA, graphNameB, uriB, usedSDDURI);
916 			
917 			// Differ between the different merge queries
918 			if ((action != null) && (action.equalsIgnoreCase("AUTO")) && (with == null) && (triples == null)) {
919 				logger.info("AUTO MERGE query detected");
920 				// Create the merged revision
921 				newRevisionNumber = MergeManagement.createMergedRevision(graphName, branchNameA, branchNameB, user, commitMessage, graphNameDiff, graphNameA, uriA, graphNameB, uriB, usedSDDURI, MergeQueryTypeEnum.AUTO, "");
922 			} else if ((action != null) && (action.equalsIgnoreCase("MANUAL")) && (with != null) && (triples != null)) {
923 				logger.info("MANUAL MERGE query detected");
924 				// Create the merged revision
925 				newRevisionNumber = MergeManagement.createMergedRevision(graphName, branchNameA, branchNameB, user, commitMessage, graphNameDiff, graphNameA, uriA, graphNameB, uriB, usedSDDURI, MergeQueryTypeEnum.MANUAL, triples);
926 			} else if ((action == null) && (with != null) && (triples != null)) {
927 				logger.info("MERGE WITH query detected");
928 				// Create the merged revision
929 				newRevisionNumber = MergeManagement.createMergedRevision(graphName, branchNameA, branchNameB, user, commitMessage, graphNameDiff, graphNameA, uriA, graphNameB, uriB, usedSDDURI, MergeQueryTypeEnum.WITH, triples);
930 			} else if ((action == null) && (with == null) && (triples == null)) {
931 				logger.info("MERGE query detected");
932 				// Check if difference model contains conflicts
933 				String queryASK = String.format(
934 						  "ASK { %n"
935 						+ "	GRAPH <%s> { %n"
936 						+ " 	?ref <http://eatld.et.tu-dresden.de/sddo#isConflicting> \"true\"^^<http://www.w3.org/2001/XMLSchema#boolean> . %n"
937 						+ "	} %n"
938 						+ "}", graphNameDiff);
939 				if (TripleStoreInterfaceSingleton.get().executeAskQuery(queryASK)) {
940 					// Difference model contains conflicts
941 					// Return the conflict model to the client
942 					responseBuilder = Response.status(Response.Status.CONFLICT);
943 					responseBuilder.entity(RevisionManagement.getContentOfGraphByConstruct(graphNameDiff, format));
944 				} else {
945 					// Difference model contains no conflicts
946 					// Create the merged revision
947 					newRevisionNumber = MergeManagement.createMergedRevision(graphName, branchNameA, branchNameB, user, commitMessage, graphNameDiff, graphNameA, uriA, graphNameB, uriB, usedSDDURI, MergeQueryTypeEnum.COMMON, "");
948 				}
949 			} else {
950 				throw new InternalErrorException("This is not a valid MERGE query: " + sparqlQuery);
951 			}
952 			
953 			String graphNameHeader;
954 			try {
955 				graphNameHeader = URLEncoder.encode(graphName, "UTF-8");
956 			} catch (UnsupportedEncodingException e) {
957 				e.printStackTrace();
958 				graphNameHeader = graphName;
959 			}
960 			
961 			
962 			// Return the revision number which were used (convert tag or branch identifier to revision number)
963 			responseBuilder.header(graphNameHeader + "-revision-number-of-branch-A", RevisionManagement.getRevisionNumber(graphName, branchNameA));
964 			responseBuilder.header(graphNameHeader + "-revision-number-of-branch-B", RevisionManagement.getRevisionNumber(graphName, branchNameB));			
965 			
966 			if (newRevisionNumber != null) {
967 				// Respond with next revision number
968 		    	responseBuilder.header(graphNameHeader + "-revision-number", newRevisionNumber);
969 				responseBuilder.header(graphNameHeader + "-revision-number-of-MASTER", RevisionManagement.getMasterRevisionNumber(graphName));
970 				logger.debug("Respond with new revision number " + newRevisionNumber + ".");
971 			}
972 		}
973 		if (!foundEntry)
974 			throw new InternalErrorException("Error in query: " + sparqlQuery);
975 		
976 		return responseBuilder.build();	
977 	}
