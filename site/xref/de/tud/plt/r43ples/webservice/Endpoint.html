358 	/**
359 	 * @param format
360 	 * 			requested mime type 
361 	 * @param sparqlQuery
362 	 * 			string containing the SPARQL query
363 	 * @return HTTP response of evaluating the sparql query 
364 	 * @throws InternalErrorException
365 	 */
366 	private Response getSparqlResponse(final String format, String sparqlQuery) throws InternalErrorException {
367 		logger.info("SPARQL query was requested. Query: " + sparqlQuery);
368 		String user = null;
369 		Matcher userMatcher = patternUser.matcher(sparqlQuery);
370 		if (userMatcher.find()) {
371 			user = userMatcher.group("user");
372 			sparqlQuery = userMatcher.replaceAll("");
373 		}
374 		String message = null;
375 		Matcher messageMatcher = patternCommitMessage.matcher(sparqlQuery);
376 		if (messageMatcher.find()) {
377 			message = messageMatcher.group("message");
378 			sparqlQuery = messageMatcher.replaceAll("");
379 		}
380 
381 		if (patternSelectAskConstructQuery.matcher(sparqlQuery).find()) {
382 			return getSelectConstructAskResponse(sparqlQuery, format);
383 		}
384 		if (patternUpdateQuery.matcher(sparqlQuery).find()) {
385 			return getUpdateResponse(sparqlQuery, user, message, format);
386 		}
387 		if (patternCreateGraph.matcher(sparqlQuery).find()) {
388 			return getCreateGraphResponse(sparqlQuery, format);
389 		}
390 		if (patternMergeQuery.matcher(sparqlQuery).find()) {
391 			return getMergeResponse(sparqlQuery, user, message, format);
392 		}
393 		if (patternDropGraph.matcher(sparqlQuery).find()) {
394 			return getDropGraphResponse(sparqlQuery, format);
395 		}
396 		if (patternBranchOrTagQuery.matcher(sparqlQuery).find()) {
397 			return getBranchOrTagResponse(sparqlQuery, user, message, format);
398 		}
399 		throw new QueryErrorException("No R43ples query detected");
400 	}

456 	/**
457 	 * Produce the response for a SELECT, ASK or CONSTRUCT SPARQL query. can handle multiple
458 	 * graphs
459 	 * 
460 	 * @param query
461 	 *            the SPARQL query
462 	 * @param format
463 	 *            the result format
464 	 * @return the response with HTTP header for every graph (revision number
465 	 *         and MASTER revision number)
466 	 * @throws InternalErrorException 
467 	 */
468 	private Response getSelectConstructAskResponse(final String query, final String format) throws InternalErrorException {
469 		ResponseBuilder responseBuilder = Response.ok();
470 		String result;
471 		
472 		if (query.contains("OPTION r43ples:SPARQL_JOIN")) {
473 			String query_rewritten = query.replace("OPTION r43ples:SPARQL_JOIN", "");
474 			query_rewritten = SparqlRewriter.rewriteQuery(query_rewritten);
475 			result = TripleStoreInterfaceSingleton.get().executeSelectConstructAskQuery(query_rewritten, format);
476 		}
477 		else {
478 			result = getSelectConstructAskResponseClassic(query, format);
479 		}
480 		
481 		if (format.equals("text/html")){
482 			responseBuilder.entity(getHTMLResult(result, query));
483 		} else {
484 			responseBuilder.entity(result);
485 		}
486 		responseBuilder.type(format);
487 		responseBuilder.header("r43ples-revisiongraph", RevisionManagement.getResponseHeaderFromQuery(query));		
488 		return responseBuilder.build();
489 	}

508 	/**
509 	 * @param query
510 	 * @param format
511 	 * @return
512 	 * @throws InternalErrorException 
513 	 */
514 	private String getSelectConstructAskResponseClassic(final String query, final String format) throws InternalErrorException {
515 		String queryM = query;
516 
517 		Matcher m = patternSelectFromPart.matcher(queryM);
518 		while (m.find()) {
519 			String graphName = m.group("graph");
520 			String type = m.group("type");
521 			String revisionNumber = m.group("revision").toLowerCase();
522 			String newGraphName;
523 
524 			// if no revision number is declared use the MASTER as default
525 			if (revisionNumber == null) {
526 				revisionNumber = "master";
527 			}
528 			if (revisionNumber.equalsIgnoreCase("master")) {
529 				// Respond with MASTER revision - nothing to be done - MASTER revisions are already created in the named graphs
530 				newGraphName = graphName;
531 			} else {
532 				if (RevisionManagement.isBranch(graphName, revisionNumber)) {
533 					newGraphName = RevisionManagement.getReferenceGraph(graphName, revisionNumber);
534 				} else {
535 					// Respond with specified revision, therefore the revision must be generated - saved in graph <RM-TEMP-graphName>
536 					newGraphName = graphName + "-temp";
537 					RevisionManagement.generateFullGraphOfRevision(graphName, revisionNumber, newGraphName);
538 				}
539 			}
540 
541 			queryM = m.replaceFirst(type + " <" + newGraphName + ">");
542 			m = patternSelectFromPart.matcher(queryM);
543 			
544 		}
545 		String response = TripleStoreInterfaceSingleton.get().executeSelectConstructAskQuery(queryM, format);
546 		return response;
547 	}

551 	 * Produce the response for a INSERT or DELETE SPARQL query.
552 	 * 
553 	 * @param query
554 	 *            the SPARQL query containing the update request
555 	 * @param user
556 	 * 			  user committing the update
557 	 * @param commitMessage
558 	 * 			  message describing the commit
559 	 * @param format
560 	 *            the result format
561 	 * @return the response with HTTP header for every graph (revision number
562 	 *         and MASTER revision number)
563 	 * @throws InternalErrorException 
564 	 */
565 	private Response getUpdateResponse(final String query, final String user, final String commitMessage,
566 			final String format) throws InternalErrorException {
567 		logger.info("Update detected");
568 		
569 		// write to add and delete sets
570 		// (replace graph names in query)
571 		String queryM = query;
572 		Matcher m = patternUpdateRevision.matcher(queryM);
573 		while (m.find()) {
574 			String graphName = m.group("graph");
575 			String revisionName = m.group("revision").toLowerCase(); 	// can contain revision
576 																		// numbers or reference
577 																		// names
578 			String action = m.group("action");
579 			String data = m.group("data");
580 			if (data == null)
581 				data = "";
582 			
583 			String newRevisionNumber = RevisionManagement.getNextRevisionNumber(graphName, revisionName);
584 			String addSetGraphUri = graphName + "-delta-added-" + newRevisionNumber;
585 			String removeSetGraphUri = graphName + "-delta-removed-" + newRevisionNumber;
586 			if (!RevisionManagement.isBranch(graphName, revisionName)) {
587 				throw new InternalErrorException("Revision is not referenced by a branch");
588 			}
589 			if (action.equalsIgnoreCase("INSERT")) {
590 				queryM = m.replaceFirst(String.format("INSERT %s { GRAPH <%s>", data, addSetGraphUri));
591 			} else if (action.equalsIgnoreCase("DELETE")) {
592 				queryM = m.replaceFirst(String.format("INSERT %s { GRAPH <%s>", data, removeSetGraphUri));
593 			} else if (action.equalsIgnoreCase("WHERE")) {
594 				// TODO ersetze mit SPARQL JOIN
595 				String tempGraphName = graphName + "-temp";
596 				RevisionManagement.generateFullGraphOfRevision(graphName, revisionName, tempGraphName);
597 				queryM = m.replaceFirst(String.format("WHERE { GRAPH <%s>", tempGraphName));
598 			}
599 			m = patternUpdateRevision.matcher(queryM);
600 		}
601 		
602 		// Remove empty insert clauses which otherwise will lead to errors
603 		m= patternEmptyGraphPattern.matcher(queryM);
604 		queryM = m.replaceAll("");
605 
606 		TripleStoreInterfaceSingleton.get().executeUpdateQuery(queryM);
607 
608 		queryM = query;
609 		m = patternGraphWithRevision.matcher(queryM);
610 		while (m.find()) {
611 			String graphName = m.group("graph");
612 			String revisionName = m.group("revision").toLowerCase();	// can contain revision
613 																		// numbers or reference
614 																		// names
615 			// General variables
616 			String newRevisionNumber = RevisionManagement.getNextRevisionNumber(graphName, revisionName);
617 			String referenceFullGraph = RevisionManagement.getReferenceGraph(graphName, revisionName);
618 			String addSetGraphUri = graphName + "-delta-added-" + newRevisionNumber;
619 			String removeSetGraphUri = graphName + "-delta-removed-" + newRevisionNumber;
620 
621 			// remove doubled data
622 			// (already existing triples in add set; not existing triples in
623 			// delete set)
624 			TripleStoreInterfaceSingleton.get().executeUpdateQuery(String.format(
625 							"DELETE { GRAPH <%s> { ?s ?p ?o. } } WHERE { GRAPH <%s> { ?s ?p ?o. } }", addSetGraphUri,
626 							referenceFullGraph));
627 			TripleStoreInterfaceSingleton.get().executeUpdateQuery(String.format(
628 					"DELETE { GRAPH <%s> { ?s ?p ?o. } } WHERE { GRAPH <%s> { ?s ?p ?o. } MINUS { GRAPH <%s> { ?s ?p ?o. } } }",
629 					removeSetGraphUri, removeSetGraphUri, referenceFullGraph));
630 
631 			// merge change sets into reference graph
632 			// (copy add set to reference graph; remove delete set from reference graph)
633 			TripleStoreInterfaceSingleton.get().executeUpdateQuery(String.format(
634 						"INSERT { GRAPH <%s> { ?s ?p ?o. } } WHERE { GRAPH <%s> { ?s ?p ?o. } }",
635 						referenceFullGraph,	addSetGraphUri));
636 			TripleStoreInterfaceSingleton.get().executeUpdateQuery(String.format(
637 					"DELETE { GRAPH <%s> { ?s ?p ?o. } } WHERE { GRAPH <%s> { ?s ?p ?o. } }", 
638 					referenceFullGraph,	removeSetGraphUri));
639 
640 			// add meta information to R43ples
641 			ArrayList<String> usedRevisionNumber = new ArrayList<String>();
642 			usedRevisionNumber.add(revisionName);
643 			RevisionManagement.addMetaInformationForNewRevision(graphName, user, commitMessage, usedRevisionNumber,
644 					newRevisionNumber, addSetGraphUri, removeSetGraphUri);
645 			
646 			
647 			queryM = m.replaceAll(String.format("GRAPH <%s> ", graphName));
648 			m = patternGraphWithRevision.matcher(queryM);
649 		}
650 		
651 		ResponseBuilder responseBuilder = Response.created(URI.create(""));
652 		
653 		String result = "Query successfully executed";
654 		if (format.equals("text/html")){
655 			responseBuilder.entity(getHTMLResult(result, query));
656 		} else {
657 			responseBuilder.entity(result);
658 		}
659 		responseBuilder.type(format);
660 		responseBuilder.header("r43ples-revisiongraph", RevisionManagement.getResponseHeaderFromQuery(query));		
661 		return responseBuilder.build();
662 	}

794 	/** 
795 	 * Creates a merge between the specified branches.
796 	 * 
797 	 * Using command: MERGE GRAPH <graphURI> BRANCH "branchNameA" INTO "branchNameB"
798 	 * 
799 	 * @param sparqlQuery the SPARQL query
800 	 * @param format the result format
801 	 * @throws InternalErrorException 
802 	 */
803 	private Response getMergeResponse(final String sparqlQuery, final String user, final String commitMessage, final String format) throws InternalErrorException {
804 		ResponseBuilder responseBuilder = Response.created(URI.create(""));
805 		logger.info("Merge creation detected");
806 
807 		// Add R43ples information
808 		Matcher m = patternMergeQuery.matcher(sparqlQuery);
809 		
810 		boolean foundEntry = false;
811 		while (m.find()) {
812 			foundEntry = true;
813 			String newRevisionNumber = null;
814 			
815 			String action = m.group("action");
816 			String graphName = m.group("graph");
817 			String sdd = m.group("sdd");
818 			String sddURI = m.group("sddURI");
819 			String branchNameA = m.group("branchNameA").toLowerCase();
820 			String branchNameB = m.group("branchNameB").toLowerCase();
821 			String with = m.group("with");
822 			String triples = m.group("triples");
823 			
824 			String revisionUriA = RevisionManagement.getRevisionUri(graphName, branchNameA);
825 			String revisionUriB = RevisionManagement.getRevisionUri(graphName, branchNameB);
826 			
827 			logger.debug("action: " + action);
828 			logger.debug("graph: " + graphName);
829 			logger.debug("sdd: " + sdd);
830 			logger.debug("sddURI: " + sddURI);
831 			logger.debug("branchNameA: " + branchNameA);
832 			logger.debug("branchNameB: " + branchNameB);
833 			logger.debug("with: " + with);
834 			logger.debug("triples: " + triples);
835 			
836 			if (!RevisionManagement.checkGraphExistence(graphName)){
837 				logger.error("Graph <"+graphName+"> does not exist.");
838 				throw new InternalErrorException("Graph <"+graphName+"> does not exist.");
839 			}
840 				
841 			
842 			// Check if A and B are different revisions
843 			if (RevisionManagement.getRevisionNumber(graphName, branchNameA).equals(RevisionManagement.getRevisionNumber(graphName, branchNameB))) {
844 				// Branches are equal - throw error
845 				throw new InternalErrorException("Specified branches are equal: " + sparqlQuery);
846 			}
847 			
848 			// Check if both are terminal nodes
849 			if (!(RevisionManagement.isBranch(graphName, branchNameA) && RevisionManagement.isBranch(graphName, branchNameB))) {
850 				throw new InternalErrorException("Non terminal nodes were used: " + sparqlQuery);
851 			}
852 
853 			
854 			// Differ between MERGE query with specified SDD and without SDD			
855 			String usedSDDURI = null;
856 			if (sdd != null) {
857 				// Specified SDD
858 				usedSDDURI = sddURI;
859 			} else {
860 				// Default SDD
861 				// Query the referenced SDD
862 				String querySDD = String.format(
863 						  "PREFIX sddo: <http://eatld.et.tu-dresden.de/sddo#> %n"
864 						+ "PREFIX rmo: <http://eatld.et.tu-dresden.de/rmo#> %n"
865 						+ "SELECT ?defaultSDD %n"
866 						+ "WHERE { GRAPH <%s> {	%n"
867 						+ "	<%s> a rmo:Graph ;%n"
868 						+ "		sddo:hasDefaultSDD ?defaultSDD . %n"
869 						+ "} }", Config.revision_graph, graphName);
870 				
871 				ResultSet resultSetSDD = TripleStoreInterfaceSingleton.get().executeSelectQuery(querySDD);
872 				if (resultSetSDD.hasNext()) {
873 					QuerySolution qs = resultSetSDD.next();
874 					usedSDDURI = qs.getResource("?defaultSDD").toString();
875 				} else {
876 					throw new InternalErrorException("Error in revision graph! Selected graph <" + graphName + "> has no default SDD referenced.");
877 				}
878 			}
879 
880 			// Get the common revision with shortest path
881 			String commonRevision = MergeManagement.getCommonRevisionWithShortestPath(revisionUriA, revisionUriB);
882 			
883 			// Create the revision progress for A and B
884 			String graphNameA = graphName + "-RM-REVISION-PROGRESS-A";
885 			String graphNameB = graphName + "-RM-REVISION-PROGRESS-B";
886 			String graphNameDiff = graphName + "-RM-DIFFERENCE-MODEL";
887 			String uriA = "http://eatld.et.tu-dresden.de/branch-A";
888 			String uriB = "http://eatld.et.tu-dresden.de/branch-B";
889 			
890 			MergeManagement.createRevisionProgress(MergeManagement.getPathBetweenStartAndTargetRevision(commonRevision, revisionUriA), graphNameA, uriA);
891 			MergeManagement.createRevisionProgress(MergeManagement.getPathBetweenStartAndTargetRevision(commonRevision, revisionUriB), graphNameB, uriB);
892 			
893 			// Create difference model
894 			MergeManagement.createDifferenceTripleModel(graphName,  graphNameDiff, graphNameA, uriA, graphNameB, uriB, usedSDDURI);
895 			
896 			// Differ between the different merge queries
897 			if ((action != null) && (action.equalsIgnoreCase("AUTO")) && (with == null) && (triples == null)) {
898 				logger.info("AUTO MERGE query detected");
899 				// Create the merged revision
900 				newRevisionNumber = MergeManagement.createMergedRevision(graphName, branchNameA, branchNameB, user, commitMessage, graphNameDiff, graphNameA, uriA, graphNameB, uriB, usedSDDURI, MergeQueryTypeEnum.AUTO, "");
901 			} else if ((action != null) && (action.equalsIgnoreCase("MANUAL")) && (with != null) && (triples != null)) {
902 				logger.info("MANUAL MERGE query detected");
903 				// Create the merged revision
904 				newRevisionNumber = MergeManagement.createMergedRevision(graphName, branchNameA, branchNameB, user, commitMessage, graphNameDiff, graphNameA, uriA, graphNameB, uriB, usedSDDURI, MergeQueryTypeEnum.MANUAL, triples);
905 			} else if ((action == null) && (with != null) && (triples != null)) {
906 				logger.info("MERGE WITH query detected");
907 				// Create the merged revision
908 				newRevisionNumber = MergeManagement.createMergedRevision(graphName, branchNameA, branchNameB, user, commitMessage, graphNameDiff, graphNameA, uriA, graphNameB, uriB, usedSDDURI, MergeQueryTypeEnum.WITH, triples);
909 			} else if ((action == null) && (with == null) && (triples == null)) {
910 				logger.info("MERGE query detected");
911 				// Check if difference model contains conflicts
912 				String queryASK = String.format(
913 						  "ASK { %n"
914 						+ "	GRAPH <%s> { %n"
915 						+ " 	?ref <http://eatld.et.tu-dresden.de/sddo#isConflicting> \"true\"^^<http://www.w3.org/2001/XMLSchema#boolean> . %n"
916 						+ "	} %n"
917 						+ "}", graphNameDiff);
918 				if (TripleStoreInterfaceSingleton.get().executeAskQuery(queryASK)) {
919 					// Difference model contains conflicts
920 					// Return the conflict model to the client
921 					responseBuilder = Response.status(Response.Status.CONFLICT);
922 					responseBuilder.entity(RevisionManagement.getContentOfGraphByConstruct(graphNameDiff));
923 				} else {
924 					// Difference model contains no conflicts
925 					// Create the merged revision
926 					newRevisionNumber = MergeManagement.createMergedRevision(graphName, branchNameA, branchNameB, user, commitMessage, graphNameDiff, graphNameA, uriA, graphNameB, uriB, usedSDDURI, MergeQueryTypeEnum.COMMON, "");
927 				}
928 			} else {
929 				throw new InternalErrorException("This is not a valid MERGE query: " + sparqlQuery);
930 			}
931 			
932 			String graphNameHeader;
933 			try {
934 				graphNameHeader = URLEncoder.encode(graphName, "UTF-8");
935 			} catch (UnsupportedEncodingException e) {
936 				e.printStackTrace();
937 				graphNameHeader = graphName;
938 			}
939 			
940 			
941 			// Return the revision number which were used (convert tag or branch identifier to revision number)
942 			responseBuilder.header(graphNameHeader + "-revision-number-of-branch-A", RevisionManagement.getRevisionNumber(graphName, branchNameA));
943 			responseBuilder.header(graphNameHeader + "-revision-number-of-branch-B", RevisionManagement.getRevisionNumber(graphName, branchNameB));			
944 			
945 			if (newRevisionNumber != null) {
946 				// Respond with next revision number
947 		    	responseBuilder.header(graphNameHeader + "-revision-number", newRevisionNumber);
948 				responseBuilder.header(graphNameHeader + "-revision-number-of-MASTER", RevisionManagement.getMasterRevisionNumber(graphName));
949 				logger.debug("Respond with new revision number " + newRevisionNumber + ".");
950 			}
951 		}
952 		if (!foundEntry)
953 			throw new InternalErrorException("Error in query: " + sparqlQuery);
954 		
955 		return responseBuilder.build();	
956 	}
