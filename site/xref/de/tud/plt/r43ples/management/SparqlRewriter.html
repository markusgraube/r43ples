46  	/** The logger. **/
47  	private static final Logger logger = Logger.getLogger(RevisionManagement.class);
48  
49  	private static final String rmo = "http://eatld.et.tu-dresden.de/rmo#";
50  	private static final String prov = "http://www.w3.org/ns/prov#";
51  
52  	private static final Node rmo_Revision = NodeFactory.createURI(rmo + "Revision");
53  	private static final Node rmo_deltaRemoved = NodeFactory.createURI(rmo + "deltaRemoved");
54  	private static final Node rmo_deltaAdded = NodeFactory.createURI(rmo + "deltaAdded");
55  	private static final Node rmo_fullGraph = NodeFactory.createURI(rmo + "fullGraph");
56  	private static final Node rmo_references = NodeFactory.createURI(rmo + "references");
57  	private static final Node prov_wasDerivedFrom = NodeFactory.createURI(prov + "wasDerivedFrom");
58  
59  	// TODO make instance variable
60  	private static int statement_i = 0;
61  	
62  	private static final Pattern pattern = Pattern.compile("FROM\\s*<(?<graph>\\S*)>\\s*REVISION\\s*\"(?<revision>\\S*)\"", Pattern.MULTILINE + Pattern.CASE_INSENSITIVE);

66  	public static String rewriteQuery(final String query_r43ples) throws InternalErrorException {
67  
68  		statement_i = 0;
69  		
70  		ExprList expression_list_revision_path = new ExprList();
71  		ExprList expression_list_last_revision = new ExprList();
72  
73  		Matcher m = pattern.matcher(query_r43ples);
74  		String query_sparql = query_r43ples;
75  		while (m.find()) {
76  			String graphName = m.group("graph");
77  			String referenceName = m.group("revision").toLowerCase();
78  			m.reset();
79  			
80  			String revisionNumber = RevisionManagement.getRevisionNumber(graphName, referenceName);
81  
82  			Tree tree =  new Tree(graphName);
83  			LinkedList<Revision> list = tree.getPathToRevision(revisionNumber);
84  			logger.info("Path to revision: " + list.toString());
85  			expression_list_last_revision.add(ExprUtils.nodeToExpr(NodeFactory.createURI(list.get(0).getRevisionUri())));
86  			list.removeLast();
87  			for (Revision ns : list) {
88  				expression_list_revision_path.add(ExprUtils.nodeToExpr(NodeFactory.createURI(ns.getRevisionUri())));
89  			}
90  
91  			query_sparql = m.replaceFirst("");
92  			m = pattern.matcher(query_sparql);
93  		}
94  
95  		// creates the Query
96  		Query qe = QueryFactory.create(query_sparql);
97  		ElementGroup eg_orginal = (ElementGroup) qe.getQueryPattern();
98  
99  		// stores the modified elements
100 		ElementGroup eg_modified = new ElementGroup();
101 		Element el = getRewrittenElementGroup(expression_list_revision_path, expression_list_last_revision,
102 				eg_orginal);
103 		eg_modified.addElement(el);
104 
105 		qe.setDistinct(true);
106 		qe.setQueryPattern(eg_modified);
107 		query_sparql = qe.serialize();
108 		logger.info("Rewritten query: \n" + query_sparql);
109 		return query_sparql;
110 	}
111 
112 	/**
113 	 * @param expression_list_revision_path
114 	 * @param expression_list_last_revision
115 	 * @param eg_orginal
116 	 * @return rewritten element group
117 	 */
118 	private static Element getRewrittenElementGroup(ExprList expression_list_revision_path,
119 			ExprList expression_list_last_revision, ElementGroup eg_orginal) {
120 		ElementGroup eg_modified = new ElementGroup();
121 		for (Element element : eg_orginal.getElements()) {
122 			try {
123 				if (element.getClass().equals(ElementMinus.class)) {
124 					ElementMinus elementMinus = (ElementMinus) element;
125 					ElementGroup elementgroup = (ElementGroup) elementMinus.getMinusElement();
126 					Element minusPart = getRewrittenElementGroup(expression_list_revision_path,
127 							expression_list_last_revision, elementgroup);
128 					ElementMinus em = new ElementMinus(minusPart);
129 					eg_modified.addElement(em);
130 				} else {
131 					ElementPathBlock epb = (ElementPathBlock) element;
132 					Iterator<TriplePath> itPatternElts = epb.patternElts();
133 
134 					while (itPatternElts.hasNext()) {
135 						TriplePath triplePath = itPatternElts.next();
136 						statement_i += 1;
137 						Element el = getRewrittenTriplePath(expression_list_revision_path,
138 								expression_list_last_revision, triplePath);
139 						eg_modified.addElement(el);
140 					}
141 				}
142 			} catch (ClassCastException e) {
143 				eg_modified.addElement(element);
144 			}
145 		}
146 		return eg_modified;
147 	}

149 	/**
150 	 * @param expression_list_revision_path
151 	 * @param expression_list_last_revision
152 	 * @param triplePath
153 	 * @return rewritten triple path element
154 	 */
155 	private static Element getRewrittenTriplePath(ExprList expression_list_revision_path,
156 			ExprList expression_list_last_revision, TriplePath triplePath) {
157 		{
158 			ElementGroup eg_modified = new ElementGroup();
159 			
160 			
161 			Node g_revision = NodeFactory.createURI(Config.revision_graph);
162 			Node var_g_delete_set_full_graph = Var.alloc("g_delete_set_full_graph_" + statement_i);
163 			Node var_reference = Var.alloc("reference_" + statement_i);
164 			Node var_r_fullgraph = Var.alloc("r_fullgraph" + statement_i);
165 			Node var_r_delete_set = Var.alloc("r_delete_set_" + statement_i);
166 			Node var_r_add_set = Var.alloc("r_add_set_" + statement_i);
167 			Node var_g_add_set = Var.alloc("g_add_set_" + statement_i);
168 			Node var_r_delta_set_old = Var.alloc("r_delta_set_old_" + statement_i);
169 			Node var_g_delta_set_old = Var.alloc("g_delta_set_old_" + statement_i);
170 
171 			ElementTriplesBlock block_triple_path = new ElementTriplesBlock();
172 			block_triple_path.addTriple(triplePath.asTriple());
173 			
174 			ElementNamedGraph ng = new ElementNamedGraph(var_g_delete_set_full_graph, block_triple_path);
175 
176 			ElementGroup eg_fullgraph = new ElementGroup();
177 			eg_fullgraph.addTriplePattern(new Triple(var_reference, rmo_references, var_r_fullgraph));
178 			eg_fullgraph.addTriplePattern(new Triple(var_reference, rmo_fullGraph, var_g_delete_set_full_graph));
179 			eg_fullgraph.addElementFilter(new ElementFilter(new E_OneOf(new ExprVar(var_r_fullgraph),
180 					expression_list_last_revision)));
181 			
182 			eg_modified.addElement(ng);
183 
184 			ElementGroup eg_union = new ElementGroup();
185 			ElementNamedGraph ng_union = new ElementNamedGraph(g_revision, eg_union);
186 			eg_modified.addElement(ng_union);
187 			
188 			eg_union.addElement(eg_fullgraph);
189 
190 			if (!expression_list_revision_path.isEmpty()) {
191 				ElementGroup eg_delete_set = new ElementGroup();
192 				eg_delete_set.addTriplePattern(new Triple(var_r_delete_set, RDF.type.asNode(), rmo_Revision));
193 				eg_delete_set.addTriplePattern(new Triple(var_r_delete_set, rmo_deltaRemoved, var_g_delete_set_full_graph));
194 				// add filter
195 				// FILTER (?rg1 IN
196 				// (<http://test.com/r43ples-dataset-revision-4>,
197 				// <http://test.com/r43ples-dataset-revision-3>))
198 				eg_delete_set.addElementFilter(new ElementFilter(new E_OneOf(new ExprVar(var_r_delete_set),
199 						expression_list_revision_path)));
200 				ElementUnion union = new ElementUnion();
201 				union.addElement(eg_delete_set);
202 
203 				// First Minus part
204 				// GRAPH ?gm1 { ?s ?p ?o }
205 				// ?rm1 a rmo:Revision.
206 				// ?rm1 rmo:deltaAdded ?gm1.
207 				// FILTER (?rm1 IN
208 				// (<http://test.com/r43ples-dataset-revision-4>,
209 				// <http://test.com/r43ples-dataset-revision-3>))
210 				ElementGroup eg_minus = new ElementGroup();
211 				ElementMinus minus = new ElementMinus(eg_minus);
212 				ElementNamedGraph ng_addset = new ElementNamedGraph(var_g_add_set, block_triple_path);
213 				eg_minus.addElement(ng_addset);
214 				ElementGroup eg_addset = new ElementGroup();
215 				ElementNamedGraph ng_addset_revision_info = new ElementNamedGraph(g_revision, eg_addset);
216 				eg_minus.addElement(ng_addset_revision_info);
217 				eg_addset.addTriplePattern(new Triple(var_r_add_set, RDF.type.asNode(), rmo_Revision));
218 				eg_addset.addTriplePattern(new Triple(var_r_add_set, rmo_deltaAdded, var_g_add_set));
219 
220 				ElementGroup eg_innerminus = new ElementGroup();
221 				// GRAPH ?gm1_old {?s ?p ?o.}
222 				// ?rm1_old a rmo:Revision.
223 				// ?rm1 prov:wasDerivedFrom+ ?rm1_old.
224 				// ?rm1_old rmo:deltaRemoved ?gm1_old.
225 				// FILTER (?rm1_old IN
226 				// (<http://test.com/r43ples-dataset-revision-4>,
227 				// <http://test.com/r43ples-dataset-revision-3>))
228 				ElementMinus inner_minus = new ElementMinus(eg_innerminus);
229 				ElementNamedGraph ng_delta_set_old = new ElementNamedGraph(var_g_delta_set_old, block_triple_path);
230 				eg_innerminus.addElement(ng_delta_set_old);
231 
232 				ElementGroup eg_ebp = new ElementGroup();
233 				ElementPathBlock ebp = new ElementPathBlock();				
234 				ebp.addTriplePath(new TriplePath(var_r_delta_set_old, new P_Link(RDF.type.asNode()), rmo_Revision));
235 				ebp.addTriplePath(new TriplePath(var_r_delta_set_old, new P_Alt(new P_Link(rmo_deltaRemoved), new P_Link(rmo_deltaAdded)), var_g_delta_set_old));
236 				ebp.addTriplePath(new TriplePath(var_r_add_set, new P_OneOrMore1(new P_Link(prov_wasDerivedFrom)),
237 						var_r_delta_set_old));
238 				eg_ebp.addElement(ebp);
239 				ElementNamedGraph ng_delta_set_old_revision_info = new ElementNamedGraph(g_revision, eg_ebp);
240 				eg_innerminus.addElement(ng_delta_set_old_revision_info);
241 
242 				eg_ebp.addElementFilter(new ElementFilter(new E_OneOf(new ExprVar(var_r_delta_set_old),
243 						expression_list_revision_path)));
244 				eg_addset.addElementFilter(new ElementFilter(new E_OneOf(new ExprVar(var_r_add_set),
245 						expression_list_revision_path)));
246 				eg_minus.addElement(inner_minus);
247 				eg_union.addElement(union);
248 				
249 				eg_modified.addElement(minus);
250 			}
251 			return eg_modified;
252 		}
253 	}
